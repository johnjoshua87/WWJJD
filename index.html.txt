    // Your pre-created quotes
    const quotes = [
      "Gotta lose money to make money",
      "What would Arteta do?",
      "Are you going to run into the fire or away from it?",
      "You know the answer - we hate Chelsea",
      "Just say yes and figure it out later"

    function newQuote() {
      const randomIndex = Math.floor(Math.random() * quotes.length);
      document.getElementById("quote-box").innerText = quotes[randomIndex];
    }
